\subsection*{第一题：单纯形法 + 对偶 + 灵敏度}
% ============================================================

\noindent 考虑线性规划：
\[
\begin{aligned}
\min\ & 2x_1 - x_2 + x_3 \\
\text{s.t.}\ & 3x_1 + x_2 + x_3 \le 6 \\
& x_1 - x_2 + 2x_3 \ge 1 \\
& x_1 + x_2 - x_3 \le 2 \\
& x_1, x_2, x_3 \ge 0
\end{aligned}
\]
\begin{enumerate}\small
  \item[(1)] 用单纯形法求解；
  \item[(2)] 写出对偶规划；
  \item[(3)] $c_1$ 由 2 改为 0，原最优解还是最优解吗？
\end{enumerate}

\reviewstep{4线性规划基本性质}{3}{4.1}{5}{4.3}
  {第1步：化成标准型}
  {引入松弛变量 $x_4,x_5$、剩余变量 $x_6$。第三个约束 $\times (-1)$ 使右端非负。
   必须确保右端 $b\ge 0$ 且所有变量 $\ge 0$。
   左图是标准型的矩阵表示，右图是非标准→标准的转化规则。\\[3pt]
   {\scriptsize 提示：$\le$ 约束 +松弛，$\ge$ 约束 $-剩余$，右 $<0$ 乘 $-1$。}}

\reviewstep{5单纯形方法}{48}{5.30}{49}{5.31}
  {第2步：大M法 — 引入人工变量}
  {第三个约束转化后缺基变量，引入人工变量 $x_7$，目标加 $+Mx_7$（极小化）。
   左图：大M法的基本思想——惩罚人工变量取非零值。
   右图：目标函数中 $M$ 的系数选取，逼迫人工变量最终为 0。\\[3pt]
   {\scriptsize 注意：两阶段法是另一选择——先 min 人工变量之和，再复原目标。}}

\reviewstep{5单纯形方法}{24}{5.10}{25}{5.11}
  {第3步：填写初始单纯形表}
  {左图：表格形式单纯形方法的结构（系数矩阵 + 目标行 + 右端）。
   右图：完整的单纯形表布局——检验数行在最底部。
   初始 BFS：$X = (0,0,0,6,0,2,1)^T$。计算 $z_j-c_j$，正检验数 $\to$ 可改进。\\[3pt]
   {\scriptsize 检验数公式：$z_j - c_j = c_B^T B^{-1} A_j - c_j$，基变量检验数恒为 0。}}

\reviewstep{5单纯形方法}{12}{5.7}{16}{5.13}
  {第4步：转轴迭代至最优}
  {左图：单纯形算法总览——进基、离基、转轴三要素。右图：收敛性定理。
   最大正检验数 $\to$ $x_3$ 进基；最小比值 $\min(6/1,\ 1/2) = 1/2$ $\to$ $x_7$ 离基。
   转轴后迭代 2--3 次，直到所有检验数 $\le 0$。最优解：$x^*=(0,\ \frac{11}{3},\ \frac{7}{3})^T$。\\[3pt]
   {\scriptsize 转轴运算：主元归一化 + 消去主列其余元素。非退化假设保证步长 $>0$。}}

\reviewstep{6对偶理论}{6}{6.4}{10}{6.8}
  {第5步：写出对偶规划}
  {左图：对称/非对称形式的对偶定义。右图：原-对偶关系对照表（极重要！）。
   原 min + 混合约束 $\to$ 对照表写出 max 形式。
   变量数 = 原约束数；$\le$ 约束 $\to$ 变量 $\ge 0$；$\ge$ 约束 $\to$ 变量 $\le 0$。\\[3pt]
   {\scriptsize 口诀：原 min → 对偶 max，原约束 → 对偶变量，原变量 → 对偶约束。}}

\reviewstep{6对偶理论}{58}{6.46}{59}{6.47}
  {第6步：灵敏度分析 — $c_1$ 变化}
  {左图：改变 $c$ 时，仅检验数和目标值受影响，右端和系数矩阵不变。
   右图：单个 $c_j$ 变化的处理——若是非基变量，只看新检验数。
   $x_1$ 是非基变量（=0），$(z_1-c_1)' = -7/3 + 2 = -1/3 \le 0$，最优解不变。}

\subsection*{第二题：灵敏度 + 对偶单纯形法}
% ============================================================

\noindent 已知初始和最优单纯形表（见真题 PDF），加入新约束 $2x_1 - x_2 + x_3 \le 9$。

\reviewstep{6对偶理论}{71}{6.57}{71}{6.57}
  {第1步：加入新约束，检查可行性}
  {将旧最优解 $x^*=(1,0,2)^T$ 代入新约束：$2\times1-0+2 = 4 \le 9$，满足。
   引入松弛变量 $x_6$，消去基变量 $x_3$ 的影响使表格规范。
   结果右端出现 $-1$ $\to$ 原可行性被破坏，需用对偶单纯形法。}

\reviewstep{6对偶理论}{25}{6.19}{26}{6.20}
  {第2步：对偶单纯形法 — 原理}
  {左图：对偶单纯形法基本思想——保持对偶可行性，逐步恢复原可行性。
   右图：与单纯形法的区别——先选离基（b 最小者），后选进基（比值最小）。
   \textbf{关键区别}：单纯形法是原可行→求对偶可行；对偶单纯形是对偶可行→求原可行。}

\reviewstep{6对偶理论}{30}{6.24}{31}{6.25}
  {第3步：进基变量选取 + 转轴求解}

   $x_6$ 离基（$b=-1$），$x_5$ 进基。转轴后右端非负，检验数 $\le 0$，得 $x^*=(3/2, 0, 1)^T$。}

\subsection*{第三题：证明 — LP 基本性质}
% ============================================================

\noindent 若标准型 LP 有非 BFS 的最优解，证它要么有 2+ 个最优 BFS，要么有一条以最优 BFS 为顶点的射线。

\reviewstep{4线性规划基本性质}{22}{4.17}{28}{4.22}
  {核心知识：极点 = BFS + 表示定理}
  {左图：极点与基本可行解等价性定理（定理 2.4）——这是 LP 几何与代数之间的桥梁。
   右图：总结——最优解必可在极点达到，求最优解 = 求最优 BFS。\\[3pt]
   {\scriptsize 证明思路：非 BFS 的 $x^*$ 可由极点凸组合 + 极方向组合表示。若只有一个最优极点 $y^*$，则 $y^* + t(y^*-x^*)$ 是射线，全部最优。}}

\reviewstepone{3凸分析}{14}{3.8}
  {支撑知识：多面体表示定理}
  {任意可行点 = 极点的凸组合 + 极方向的非负线性组合。有界多面体（多胞形）无极方向，只需极点凸组合。这是 Q3 证明的几何基础。}

\subsection*{第四题：DFP 拟牛顿法}
% ============================================================

\noindent $\min\ f(X) = x_1^2 + 2x_2^2 + x_1$，取 $X^{(0)}=(2,1)^T$，$H_0=I$。

\reviewstep{10使用导数的优化算法}{91}{10.52}{92}{10.53}
  {第1步：DFP 公式 + 算法步骤}
  {左图：DFP 校正公式——对 Hessian 逆矩阵 $H$ 的秩 2 校正。右图：算法流程。
   $\nabla f = (2x_1+1, 4x_2)^T$。$X^{(0)}$ 处 $\nabla f^{(0)}=(5,4)^T$，$d^{(0)} = -H_0 \nabla f^{(0)} = -(5,4)^T$。}

\reviewstep{9一维搜索}{51}{9.8}{52}{9.9}
  {第2步：一维搜索求步长 $\lambda_0$}
  {左图：不精确线搜索概念。右图：Armijo 准则。
   $X^{(0)} + \lambda d^{(0)} = (2-5\lambda, 1-4\lambda)$，代入 $f$ 得 $\phi(\lambda)$。
   令 $\phi'(\lambda)=0$，得 $\lambda_0 = 41/114$。
   考试中须完整写出 $\phi(\lambda)$ 表达式再求导。}

\reviewstep{10使用导数的优化算法}{95}{10.56}{96}{10.57}
  {第3步：DFP 校正矩阵计算（核心）}
  {$X^{(1)} = \frac{1}{114}(23, -50)^T$，$\nabla f^{(1)} = \frac{1}{114}(160, -200)^T$。
   $s_0 = X^{(1)}-X^{(0)}$，$y_0 = \nabla f^{(1)}-\nabla f^{(0)}$。
   代入 DFP 公式：$H_1 = H_0 + \frac{s_0 s_0^T}{s_0^T y_0} - \frac{H_0 y_0 y_0^T H_0}{y_0^T H_0 y_0}$。
   这是 DFP 的\textbf{核心得分点}——对 $H$ 做秩 2 校正，每个矩阵运算都必须正确。}

\reviewstep{10使用导数的优化算法}{102}{10.63}{113}{10.74}
  {第4步：第二次迭代 + 收敛检验}
  {$d^{(1)} = -H_1 \nabla f^{(1)}$，一维搜索得 $\lambda_1 = 5/57$。$X^{(2)} = (-1/2, 0)^T$。
   $\nabla f^{(2)} = 0$ $\to$ 驻点。Hessian $\nabla^2 f$ 正定 $\to$ 极小点。两端 DFP 收敛。\\[3pt]
   {\scriptsize 注意：DFP 有二次终止性——对正定二次函数，至多 n 轮迭代收敛。BFGS 性能通常优于 DFP。}}

\subsection*{第六题：FJ条件 + KKT + 非线性对偶}
% ============================================================

\noindent $\min\ x_1^2 + x_2^2$，s.t.\ $(x_1-1)^3 - x_2^2 = 0$。
\begin{enumerate}\small
  \item[(1)] 找出所有满足 FJ 条件的可行点；
  \item[(2)] 证不存在 KKT 点；
  \item[(3)] 消去约束法是否正确？校正；
  \item[(4)] 写对偶规划。
\end{enumerate}

\reviewstep{7最优性条件}{17}{7.9}{19}{7.11}
  {第1步：FJ 条件的方程建立}
  {左图：FJ 条件的几何前提——下降方向 + 可行方向的交集为空。
   右图：FJ 必要条件的表述。$\nabla f = (2x_1, 2x_2)^T$，$\nabla g = (3(x_1-1)^2, -2x_2)^T$。
   FJ：$\lambda_0 \nabla f + \mu \nabla g = 0$，$\lambda_0 \ge 0$，$\mu \ge 0$，不全为零。}

\reviewstep{7最优性条件}{26}{7.18}{27}{7.19}
  {第2步：KKT 条件 —— 有约束规格保证 $\lambda_0=1$}
  {左图：KKT 条件的引入——通过约束规格保证 $\lambda_0 \neq 0$（可归一化）。
   右图：Lagrange 函数定义 + KKT 条件的简洁表示。\\[3pt]
   {\scriptsize FJ 得到 $(1,0)$，代入 KKT：$2\cdot 1 + \mu\cdot 3(0)^2 = 2 \neq 0$，矛盾。\textbf{真题得分点}：必须写清楚为什么 KKT 不成立。}}

\reviewstep{7最优性条件}{48}{7.32}{51}{7.35}
  {第3步：做法的错误与校正}
  {左图：非线性对偶形式。右图：对偶实例。
   错误：消 $x_2$ 时忘了隐条件 $x_2^2 \ge 0 \Rightarrow (x_1-1)^3 \ge 0 \Rightarrow x_1 \ge 1$。
   校正：$\min\ x_1^2 + (x_1-1)^3$，s.t.\ $x_1 \ge 1$。当 $x_1 \to \infty$，$f \to \infty$；但取 $x_1=1,x_2=0$ 时 $f=1$。$x_1$ 不能 $<1$（否则无实 $x_2$）。}

\reviewstep{7最优性条件}{52}{7.36}{55}{7.39}
  {第4步：非线性对偶 + 强对偶定理}
  {左图：弱对偶定理。右图：强对偶定理（需凸性+Slater条件）。
   Lagrange 对偶：$\theta(\mu) = \inf_x\{x_1^2+x_2^2 + \mu[(x_1-1)^3 - x_2^2]\}$。
   $\mu \neq 0$ 时 $\theta(\mu) = -\infty$（取 $x_2 \to \infty$）；$\mu=0$ 时 $\theta(0)=0$。
   对偶规划 $\max\ \theta(\mu) = 0$。\textbf{非凸问题，强对偶不成立，对偶间隙非零。}}
