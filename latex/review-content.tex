\subsection*{第一题：单纯形法 + 对偶 + 灵敏度}
% ============================================================

\noindent 考虑线性规划：
\[
\begin{aligned}
\min\ & 2x_1 - x_2 + x_3 \\
\text{s.t.}\ & 3x_1 + x_2 + x_3 \le 6 \\
& x_1 - x_2 + 2x_3 \ge 1 \\
& x_1 + x_2 - x_3 \le 2 \\
& x_1, x_2, x_3 \ge 0
\end{aligned}
\]
\begin{enumerate}\small
  \item[(1)] 用单纯形法求解；
  \item[(2)] 写出对偶规划；
  \item[(3)] $c_1$ 由 2 改为 0，原最优解还是最优解吗？
\end{enumerate}

\reviewstep{4线性规划基本性质}{3}{第4章 第1张}
  {第1步：化成标准型}
  {引入松弛变量 $x_4,x_5$、剩余变量 $x_6$。第三个约束 $\times (-1)$ 使右端非负。
   标准型形如：$\min c^Tx,\ \text{s.t.}\ Ax=b,\ x\ge 0$。注意右端 $b\ge 0$。}

\reviewstep{5单纯形方法}{48}{第5章 第30张}
  {第2步：大M法 — 缺基变量时引入人工变量}
  {第三个约束转化后缺基变量，引入人工变量 $x_7$，目标加惩罚项 $Mx_7$（极小化 $+M$）。
   这是用大M法的信号：约束中没有现成的单位向量作为初始基。}

\reviewstep{5单纯形方法}{24}{第5章 第10张}
  {第3步：写初始单纯形表，计算检验数}
  {填入系数矩阵和目标行。$z_j-c_j > 0$ 表示该非基变量进基可改进目标值。
   初始基本可行解 $X=(0,0,0,6,0,2,1)^T$。}

\reviewstep{5单纯形方法}{12}{第5章 第7张}
  {第4步：转轴迭代（共2--3次）}
  {最大正检验数 $\to$ 进基；最小比值 $\min(6/1, 1/2)=1/2$ $\to$ 离基。
   转轴后 $x_7$ 出基（人工变量可删列）。继续直到所有检验数 $\le 0$。
   最优解：$x^*=(0, \frac{11}{3}, \frac{7}{3})^T$。}

\reviewstep{6对偶理论}{6}{第6章 第4张}
  {第5步：写对偶规划}
  {对照对偶关系表：原 min $\leftrightarrow$ 对偶 max。变量数 = 原约束数。
   原 $\le$ $\to$ 对偶变量 $\ge 0$；原 $\ge$ $\to$ 对偶变量 $\le 0$；原 $=$ $\to$ 无约束。}

\reviewstep{6对偶理论}{58}{第6章 第46张}
  {第6步：灵敏度分析 — $c_1$ 变化}
  {$x_1$ 在最优表中是非基变量（取值为 0），只需检查新检验数：
   $(z_1-c_1)' = -7/3 + 2 = -1/3 \le 0$，仍非正，最优解不变。
   如果新检验数 $>0$，以当前表为起点继续单纯形迭代。}

\subsection*{第二题：灵敏度 + 对偶单纯形法}
% ============================================================

\noindent 已知初始和最优单纯形表（见真题），加入新约束 $2x_1 - x_2 + x_3 \le 9$。问：原最优解还是最优解吗？

\reviewstep{6对偶理论}{71}{第6章 第57张}
  {第1步：检验旧最优解是否满足新约束}
  {将 $x^*=(1,0,2)^T$ 代入 $2x_1-x_2+x_3 = 4 \le 9$，满足。
   引入松弛变量 $x_6$，消去基变量 $x_3$ 的影响使表格规范。结果右端出现 $-1$。}

\reviewstep{6对偶理论}{25}{第6章 第19张}
  {第2步：对偶单纯形法 — 先选离基}
  {右端有负分量，原可行性被破坏。但检验数仍 $\le 0$（对偶可行），用\textbf{对偶单纯形法}。
   先定离基（最小右端值），再定进基（最小比值 $|\frac{\text{检验数}}{\text{主行元素}}|$，主行元素 $<0$）。}

\reviewstep{6对偶理论}{30}{第6章 第24张}
  {第3步：进基变量选取 + 转轴}
  {$x_6$ 离基（$b=-1$），比值 $\min(|{-2}/{-2}|,\ |{-1}/{-2}|) = 1/2$，$x_5$ 进基。
   转轴后右端非负，检验数 $\le 0$，得新最优解 $x^*=(3/2, 0, 1)^T$。}

\subsection*{第三题：证明 — LP 基本性质}
% ============================================================

\noindent 若标准型 LP 有一个非基本可行解的最优解，证它要么有 2+ 个最优 BFS，要么有一条以最优 BFS 为顶点的射线。

\reviewstep{4线性规划基本性质}{22}{第4章 第17张}
  {核心定理：极点 = 基本可行解}
  {可行域 $K=\{x|Ax=b, x\ge 0\}$ 的极点集与基本可行解集合等价。
   LP 有最优解 $\Rightarrow$ 必有最优基本可行解（在极点上达到）。}

\reviewstep{3凸分析}{14}{第3章 第8张}
  {表示定理：任一点 = 极点凸组合 + 极方向非负组合}
  {设只有一个最优 BFS $y^*$。以 $y^*$ 为顶点、$y^*-x^*$ 为方向的射线全是最优解，不含其他极点，在可行域内。命题得证。}

\subsection*{第四题：DFP 拟牛顿法}
% ============================================================

\noindent $\min\ f(X) = x_1^2 + 2x_2^2 + x_1$，取 $X^{(0)}=(2,1)^T$，$H_0=I$。

\reviewstep{10使用导数的优化算法}{91}{第10章 第52张}
  {第1步：计算梯度 + 初始搜索方向}
  {$\nabla f = (2x_1+1,\ 4x_2)^T$，在 $X^{(0)}$ 处 $\nabla f^{(0)} = (5, 4)^T$。
   $d^{(0)} = -H_0 \nabla f^{(0)} = -(5, 4)^T$。$H_0=I$ 时第一步等同最速下降方向。}

\reviewstep{9一维搜索}{51}{第9章 第8张}
  {第2步：一维搜索求步长}
  {沿 $d^{(0)}$：$X^{(0)}+\lambda d^{(0)} = (2-5\lambda,\ 1-4\lambda)$。
   代入 $f$ 得 $\phi(\lambda)$，令 $\phi'(\lambda)=0$，得 $\lambda_0 = 41/114$。
   考试必须写出 $\phi(\lambda)$ 表达式再求导。}

\reviewstep{10使用导数的优化算法}{91}{第10章 第52张}
  {第3步：DFP 校正矩阵（核心步骤）}
  {得 $X^{(1)} = \frac{1}{114}(23, -50)^T$，$\nabla f^{(1)} = \frac{1}{114}(160, -200)^T$。
   $s_0 = X^{(1)}-X^{(0)}$，$y_0 = \nabla f^{(1)}-\nabla f^{(0)}$。
   代入 DFP 公式：$H_1 = H_0 + \frac{s_0 s_0^T}{s_0^T y_0} - \frac{H_0 y_0 y_0^T H_0}{y_0^T H_0 y_0}$。}

\reviewstep{10使用导数的优化算法}{102}{第10章 第63张}
  {第4步：第二次迭代，检验收敛}
  {$d^{(1)} = -H_1 \nabla f^{(1)}$，一维搜索得 $\lambda_1 = 5/57$，$X^{(2)} = (-1/2, 0)^T$。
   $\nabla f^{(2)} = 0$ $\to$ 驻点。Hessian 正定 $\to$ 极小点。两轮迭代收敛。}

\subsection*{第五题：可行方向法（Rosen 梯度投影）}
% ============================================================

\noindent $\min\ x_1^2+2x_2^2-x_1x_2+x_3^2+x_3$，s.t.\ $x_1+x_2=6$，$-x_1+x_2+x_3\le 2$，$x_i\ge 0$。

\reviewstep{12可行方向法}{2}{第12章 第1张}
  {第1步：可行方向法基本思路}
  {从可行点出发 $\to$ 找下降可行方向 $\to$ 一维搜索 $\to$ 新可行点 $\to$ 重复直到 KKT 点。
   与无约束下降算法的区别：方向必须保持在可行域内。}

\reviewstep{12可行方向法}{42}{第12章 第36张}
  {第2步：Rosen 梯度投影 — 识别积极约束}
  {取初始可行点 $X^{(0)} = (2,2,2)^T$。积极约束：前两个取等号。
   构造 $M$（积极约束系数矩阵）。先化简：最优解处 $x_3=0$，问题降为二维。}

\reviewstep{12可行方向法}{44}{第12章 第38张}
  {第3步：计算投影矩阵 + 搜索方向}
  {$P = I - M^T(MM^T)^{-1}M$。$d^{(0)} = -P\nabla f(X^{(0)})$。
   核心：把负梯度投影到约束切空间上，保证方向可行。}

\reviewstep{12可行方向法}{47}{第12章 第41张}
  {第4步：一维搜索 + 迭代}
  {沿 $d^{(0)}$ 做一维搜索，步长 $\lambda_0 = 0.5$。$X^{(1)} = (2.75, 2, 1.25)^T$。
   第二次迭代：新积极约束，重算投影矩阵。$d=0$ 时检查 $w$ 分量 $\to$ KKT 检验。}

\subsection*{第六题：FJ条件 + KKT + 非线性对偶}
% ============================================================

\noindent $\min\ x_1^2 + x_2^2$，s.t.\ $(x_1-1)^3 - x_2^2 = 0$。
\begin{enumerate}\small
  \item[(1)] 找出所有满足 FJ 条件的可行点；
  \item[(2)] 证不存在可行点满足 KKT；
  \item[(3)] 消去约束化为一元是否正确？如不，校正；
  \item[(4)] 写对偶规划。
\end{enumerate}

\reviewstep{7最优性条件}{17}{第7章 第9张}
  {第1步：写 FJ 方程}
  {$\nabla f = (2x_1, 2x_2)^T$，$\nabla g = (3(x_1-1)^2, -2x_2)^T$。
   FJ：$\lambda_0 \nabla f + \mu \nabla g = 0$，$(\lambda_0, \mu) \neq (0,0)$。
   分 $\lambda_0=0$（得点 $(1,0)$）和 $\lambda_0 \neq 0$（讨论 $x_2=0$ 或 $\mu=1$，均无解）。}

\reviewstep{7最优性条件}{26}{第7章 第18张}
  {第2步：FJ 到 KKT — 约束规格的作用}
  {FJ 得到 $(1,0)$，但代入 KKT（$\lambda_0 \neq 0$，归一化）：
   $2x_1 + \mu \cdot 3(x_1-1)^2 = 0$，在 $(1,0)$ 处 $2=0$，矛盾。
   \textbf{关键：}FJ 允许 $\lambda_0=0$（不含目标函数），KKT 通过约束规格排除此情形。}

\reviewstep{7最优性条件}{48}{第7章 第32张}
  {第3步：做法校正}
  {消去约束漏掉了隐含条件：$x_2^2 = (x_1-1)^3 \ge 0 \Rightarrow x_1 \ge 1$。
   正确做法：$\min\ x_1^2 + (x_1-1)^3$，s.t.\ $x_1 \ge 1$。考虑边界行为判别无下界。}

\reviewstep{7最优性条件}{52}{第7章 第36张}
  {第4步：非线性对偶}
  {Lagrange 对偶 $\theta(\mu) = \inf_x \{x_1^2+x_2^2 + \mu[(x_1-1)^3 - x_2^2]\}$。
   $\mu \neq 0$ 时 $\theta(\mu)=-\infty$；$\mu=0$ 时 $\theta(0)=0$。对偶规划 $\max\ \theta(\mu) = 0$。
   弱对偶成立但强对偶不成立（非凸问题），对偶间隙非零。}
