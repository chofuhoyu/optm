\subsection*{第一题：单纯形法 + 对偶 + 灵敏度}
% ============================================================

\noindent 考虑线性规划：
\[
\begin{aligned}
\min\ & 2x_1 - x_2 + x_3 \\
\text{s.t.}\ & 3x_1 + x_2 + x_3 \le 6 \\
& x_1 - x_2 + 2x_3 \ge 1 \\
& x_1 + x_2 - x_3 \le 2 \\
& x_1, x_2, x_3 \ge 0
\end{aligned}
\]
\begin{enumerate}
  \item[(1)] 用单纯形法求解；
  \item[(2)] 写出对偶规划；
  \item[(3)] $c_1$ 由 2 改为 0，原最优解还是最优解吗？
\end{enumerate}

\reviewstep{第1步：化成标准型}
  {引入松弛变量 $x_4,x_6$ 和剩余变量 $x_5$：
  \begin{align*}
    \min\ & 2x_1 - x_2 + x_3 \\
    \text{s.t.}\ & 3x_1 + x_2 + x_3 + x_4 = 6 \\
    & x_1 - x_2 + 2x_3 - x_5 = 1 \\
    & x_1 + x_2 - x_3 + x_6 = 2 \\
    & x_j \ge 0,\ j=1,\dots,6
  \end{align*}
  三个方程右端均 $\ge 0$，满足标准型 $b\ge 0$。$x_4$、$x_6$ 系数为 $+1$ 可做基变量，但第二行 $x_5$ 系数为 $-1$ 不能做基变量，需下一步引入人工变量。}
  {标准型要求：目标 $\min$、约束 $Ax=b$、右端 $b\ge 0$、变量 $\ge 0$。$\le$ 加松弛变量（$+$），$\ge$ 减剩余变量（$-$）。第三个约束 $x_1+x_2-x_3\le 2$ 右端已是 $2\ge 0$，直接加松弛变量即得 $x_1+x_2-x_3+x_6=2$，无需乘 $-1$。第二行因剩余变量系数为 $-1$，尚缺基变量。}
  {\refslide{4线性规划基本性质}{3}{4.1}\par\vspace{4pt}\refslide{4线性规划基本性质}{5}{4.3}}\reviewstep{第2步：大M法引入人工变量}
  {第二行 $x_1-x_2+2x_3-x_5=1$ 缺基变量（$x_5$ 系数为 $-1$），引入人工变量 $x_7$：
  \begin{align*}
    \min\ & 2x_1 - x_2 + x_3 + Mx_7 \\
    \text{s.t.}\ & 3x_1 + x_2 + x_3 + x_4 = 6 \\
    & x_1 - x_2 + 2x_3 - x_5 + x_7 = 1 \\
    & x_1 + x_2 - x_3 + x_6 = 2 \\
    & x_j \ge 0,\ j=1,\dots,7
  \end{align*}
  初始基本可行解：$X = (0,0,0,6,0,2,1)^T$，基变量为 $x_4,x_7,x_6$。}
  {人工变量 $x_7$ 为第二行提供缺失的基变量。目标加 $Mx_7$（$M$ 为极大正数），迫使 $x_7\to 0$。若最优解中 $x_7\neq 0$，原问题无可行解。注意：只有 $\ge$ 约束（减剩余变量后系数为 $-1$）才需要人工变量；$\le$ 约束加松弛变量后自然提供基变量。}
  {\refslide{5单纯形方法}{48}{5.30}\par\vspace{4pt}\refslide{5单纯形方法}{49}{5.31}}\reviewstep{第3步：单纯形表迭代}
  {初始单纯形表后，$x_3$ 检验数最大（$-1+2M$），进基。最小比值 $\min(6/1,\ 1/2)=1/2$，$x_7$ 离基。转轴后删去 $x_7$ 列。第二次迭代：$x_2$ 检验数 $1/2>0$ 进基，$x_4$ 离基。最终：
  \[
  \text{所有检验数}\le 0,\quad x^* = (0,\ \tfrac{11}{3},\ \tfrac{7}{3})^T
  \]}
  {进基：最大正检验数对应非基变量。离基：最小比值 $\min\{b_i/a_{ij}: a_{ij}>0\}$。人工变量一旦离基可删列。}
  {\refslide{5单纯形方法}{24}{5.10}\par\vspace{4pt}\refslide{5单纯形方法}{16}{5.9}}\reviewstep{第4步：写出对偶规划}
  {对照原-对偶关系表（原 min + $\le$约束 $\to$ 对偶变量 $\ge 0$），得：
  \begin{align*}
    \max\ & 6w_1 + w_2 + 2w_3 \\
    \text{s.t.}\ & 3w_1 + w_2 + w_3 \le 2 \\
    & w_1 - w_2 + w_3 \le -1 \\
    & w_1 + 2w_2 - w_3 \le 1 \\
    & w_1 \le 0,\ w_2 \ge 0,\ w_3 \le 0
  \end{align*}}
  {对偶关系表是核心：原 min → 对偶 max；原约束 $\le$ → 变量 $\ge 0$；原约束 $\ge$ → 变量 $\le 0$；原变量 $\ge 0$ → 对偶约束 $\le$。}
  {\refslide{6对偶理论}{6}{6.4}\par\vspace{4pt}\refslide{6对偶理论}{10}{6.8}}\reviewstep{第5步：灵敏度分析}
  {$x_1$ 在最优表中为非基变量（取值 $0$），$c_1$ 由 $2$ 改为 $0$：
  \[
  (z_1-c_1)' = -\frac{7}{3} + 2 = -\frac{1}{3} \le 0
  \]
  检验数仍非正，故最优解不变。}
  {非基变量的 $c$ 变化：只需检查该变量对应的检验数。新检验数 $\le 0$ → 最优解不变；$>0$ → 以当前表为起点继续单纯形迭代。}
  {\refslide{6对偶理论}{58}{6.46}\par\vspace{4pt}\refslide{6对偶理论}{59}{6.47}}\rule{\textwidth}{0.4pt}

\subsection*{第二题：灵敏度 + 对偶单纯形法}
% ============================================================

\noindent 已知初始和最优单纯形表（见真题 PDF），加入新约束 $2x_1 - x_2 + x_3 \le 9$。问原解是否仍最优？

\reviewstep{加入新约束 + 对偶单纯形法}
  {旧最优解 $x^*=(1,0,2)^T$ 代入新约束：$4\le 9$，满足。引入 $x_6$，消去 $x_3$ 影响后右端出现 $-1$→原可行性破坏。检验数均 $\le 0$（对偶可行），用\textbf{对偶单纯形法}：
  $x_6$ 离基（$b=-1$），$\min(|-2/-2|,\ |-1/-2|)=1/2$，$x_5$ 进基。转轴后：
  \[
  x^* = (\tfrac{3}{2},\ 0,\ 1)^T,\quad \text{右端均}\ge 0,\quad \text{检验数均}\le 0
  \]}
  {对偶单纯形法适用于右端有负但检验数 $\le 0$ 的情形。先选离基（最小右端），后选进基（负主行元素对应检验数，取绝对值最小比值）。}

\subsection*{第三题：证明 — LP 基本性质}
% ============================================================

\noindent 若标准型 LP 有非 BFS 的最优解，证它要么有 $2+$ 个最优 BFS，要么有一条以最优 BFS 为顶点的射线。

\reviewstep{证明思路}
  {记 $x^*$ 为非 BFS 的最优解。由 LP 基本性质：存在最优 BFS $y^*$（因标准型 LP 有最优解必有最优基本可行解）。以 $y^*$ 为顶点、$y^*-x^*$ 为方向的射线上的点全为最优解（设只有一个最优极点）。此射线不含其他极点，故在可行域内，命题得证。}
  {核心定理：极点 = 基本可行解 + LP 有最优解必可在极点达到。证明题靠逻辑链：非 BFS 最优解 → 必有 BFS 最优解 → 构造射线 → 分析极点数 → 证毕。}

\subsection*{第四题：DFP 拟牛顿法}
% ============================================================

\noindent $\min\ f(X) = x_1^2 + 2x_2^2 + x_1$，$X^{(0)}=(2,1)^T$，$H_0=I$。

\reviewstep{第1步：计算梯度 + 搜索方向}
  {$\nabla f = (2x_1+1,\ 4x_2)^T$，$\nabla f^{(0)} = (5, 4)^T$。\\
  搜索方向：$d^{(0)} = -H_0 \nabla f^{(0)} = -(5, 4)^T$。}
  {DFP 法第一步：$H_0=I$ 时与最速下降方向相同。}
  {\refslide{10使用导数的优化算法}{91}{10.52}\par\vspace{4pt}\refslide{10使用导数的优化算法}{92}{10.53}}\reviewstep{第2步：一维搜索求 $\lambda_0$}
  {$X^{(0)} + \lambda d^{(0)} = (2-5\lambda,\ 1-4\lambda)$。代入目标函数：
  \[
  \phi(\lambda) = (2-5\lambda)^2 + 2(1-4\lambda)^2 + (2-5\lambda)
  \]
  令 $\phi'(\lambda)=0$，得 $\lambda_0 = 41/114$。
  $X^{(1)} = \frac{1}{114}(23, -50)^T$，
  $\nabla f^{(1)} = \frac{1}{114}(160, -200)^T$。}
  {必须写出 $\phi(\lambda)$ 表达式再求导，这是考试得分点。}
  {\refslide{9一维搜索}{51}{9.8}\par\vspace{4pt}\refslide{9一维搜索}{52}{9.9}}\reviewstep{第3步：DFP 校正（核心得分点）}
  {$s_0 = X^{(1)}-X^{(0)} = -\frac{41}{114}(5,4)^T$，$y_0 = \nabla f^{(1)}-\nabla f^{(0)} = -\frac{41}{114}(10,16)^T$。\\
  代入 DFP 公式：
  \[
  H_1 = H_0 + \frac{s_0 s_0^T}{s_0^T y_0} - \frac{H_0 y_0 y_0^T H_0}{y_0^T H_0 y_0}
  \]
  秩2校正后 $H_1$ 不再为单位阵。}
  {DFP 校的是 $H$（Hessian 逆）。公式记牢：两项修正——第一项正定修正，第二项消去旧曲率信息。}
  {\refslide{10使用导数的优化算法}{95}{10.56}\par\vspace{4pt}\refslide{10使用导数的优化算法}{96}{10.57}}\reviewstep{第4步：第二次迭代 + 收敛}
  {$d^{(1)} = -H_1 \nabla f^{(1)}$。一维搜索得 $\lambda_1 = 5/57$。\\
  $X^{(2)} = (-\frac{1}{2},\ 0)^T$，$\nabla f^{(2)} = 0$ → 驻点。\\
  Hessian $\nabla^2 f = \begin{pmatrix}2&0\\0&4\end{pmatrix}$ 正定 → 极小点。两轮迭代收敛。}
  {DFP 有二次终止性：对正定二次函数至多 $n$ 轮收敛。此题 $n=2$，恰好两轮到达。}
  {\refslide{10使用导数的优化算法}{102}{10.63}\par\vspace{4pt}\refslide{10使用导数的优化算法}{113}{10.74}}\rule{\textwidth}{0.4pt}

\subsection*{第五题：可行方向法（Rosen 梯度投影）}
% ============================================================

\noindent $\min\ x_1^2+2x_2^2-x_1x_2+x_3^2+x_3$，s.t.\ $x_1+x_2=6$，$-x_1+x_2+x_3\le 2$，$x_i\ge 0$。

\reviewstep{第1步：化简问题 + 取初始点}
  {最优解处 $x_3=0$，问题化为二维。取可行点 $X^{(0)} = (2,2,2)^T$，前两约束取等号。
  \[
  \nabla f = \begin{pmatrix} 2x_1-x_2 \\ 4x_2-x_1 \\ 2x_3+1 \end{pmatrix},\quad
  \nabla f^{(0)} = \begin{pmatrix} 2 \\ 6 \\ 5 \end{pmatrix}
  \]
  积极约束系数矩阵：$M = \begin{pmatrix}1&1&0\\-1&1&0\end{pmatrix}$}
  {可行方向法从可行域内部或边界出发，找下降+可行方向。积极约束 = 当前取等号的不等式约束。}
  {\refslide{12可行方向法}{2}{12.1}\par\vspace{4pt}\refslide{12可行方向法}{3}{12.2}}\reviewstep{第2步：计算投影矩阵 + 搜索方向}
  {$P = I - M^T(MM^T)^{-1}M$，代入计算：
  \[
  P = \begin{pmatrix}0.5&0&-0.5\\0&0&0\\-0.5&0&0.5\end{pmatrix}
  \]
  $d^{(0)} = -P\nabla f^{(0)} = (1.5,\ 0,\ -1.5)^T$。}
  {投影矩阵 $P$ 把负梯度投影到约束切空间上。$P$ 的特征根为 0 或 1。若 $d=0$ 且所有 $w_i\ge 0$ → KKT 点。}

  $X^{(1)} = (2.75,\ 2,\ 1.25)^T$。
  第二次迭代：$\nabla f^{(1)}$，$d^{(1)}=0$，需调整 $M$ 重新计算。}

\subsection*{第六题：FJ条件 + KKT + 非线性对偶}
% ============================================================

\noindent $\min\ x_1^2 + x_2^2$，s.t.\ $(x_1-1)^3 - x_2^2 = 0$。
\begin{enumerate}
  \item[(1)] 找出所有满足 FJ 条件的可行点；
  \item[(2)] 证明不存在 KKT 点；
  \item[(3)] 消去约束化为一元是否正确？校正；
  \item[(4)] 写对偶规划。
\end{enumerate}

\reviewstep{第1步：FJ 方程求解}
  {$\nabla f = (2x_1, 2x_2)^T$，$\nabla g = (3(x_1-1)^2,\ -2x_2)^T$。FJ 方程：
  \[
  \lambda_0 \begin{pmatrix}2x_1\\2x_2\end{pmatrix}
  + \mu \begin{pmatrix}3(x_1-1)^2\\-2x_2\end{pmatrix} = 0,\quad
  \lambda_0\ge 0,\ \mu\ge 0,\ (\lambda_0,\mu)\neq(0,0)
  \]
  $\lambda_0=0$ 时 $\mu\neq 0$ → $x_1=1,\ x_2=0$。\\
  $\lambda_0\neq 0$ 时 $(1-\mu)x_2=0$ → $\mu=1$（无实根）或 $x_2=0$（不满足FJ）。
  故仅有 FJ 点：$(1,0)$。}
  {FJ 条件允许 $\lambda_0=0$（目标函数信息缺失）。分情形讨论是关键技巧。}
  {\refslide{7最优性条件}{17}{7.9}\par\vspace{4pt}\refslide{7最优性条件}{19}{7.11}}\reviewstep{第2步：KKT 不存在}
  {KKT：$\lambda_0\neq 0$（归一化为 1）。将 $(1,0)$ 代入：
  \[
  \begin{pmatrix}2\cdot 1\\2\cdot 0\end{pmatrix}
  + \mu\begin{pmatrix}3(0)^2\\-2\cdot 0\end{pmatrix} = \begin{pmatrix}2\\0\end{pmatrix} \neq 0
  \]
  矛盾，故不存在 KKT 点。}
  {KKT 要求 $\lambda_0\neq 0$（约束规格保证）。核心区别：FJ 有解但 KKT 没解 → 因为约束不满足约束规格。}
  {\refslide{7最优性条件}{26}{7.18}\par\vspace{4pt}\refslide{7最优性条件}{27}{7.19}}\reviewstep{第3步：做法的错误与校正}
  {消去 $x_2^2 = (x_1-1)^3$ 代入目标得 $\min\ x_1^2 + (x_1-1)^3$。
  错误：遗漏了隐条件 $(x_1-1)^3 = x_2^2 \ge 0 \Rightarrow x_1 \ge 1$。
  校正：加上 $x_1 \ge 1$ 再求解。$x_1^2+(x_1-1)^3$ 在 $[1,\infty)$ 上最小值在 $x_1=1$ 处，得 $f=1$，$x^*=(1,0)^T$。}
  {消元法必须保留隐含约束。$x^2 \ge 0$ 这类条件容易被忽略。}
  {\refslide{7最优性条件}{48}{7.32}\par\vspace{4pt}\refslide{7最优性条件}{51}{7.35}}\reviewstep{第4步：非线性对偶}
  {Lagrange 对偶函数：
  \[
  \theta(\mu) = \inf_{x}\{x_1^2+x_2^2 + \mu[(x_1-1)^3 - x_2^2]\}
  = \begin{cases}
  0, & \mu = 0 \\
  -\infty, & \mu \neq 0
  \end{cases}
  \]
  对偶规划：$\max\ \theta(\mu) = 0$。\\
  注意：弱对偶成立（$0 \le 1$），但强对偶不成立（非凸问题，对偶间隙 $=1$）。}
  {非凸问题的对偶：弱对偶总成立，强对偶不一定。对偶函数可能为 $-\infty$（无下界），此时对偶无实际意义。}
  {\refslide{7最优性条件}{52}{7.36}\par\vspace{4pt}\refslide{7最优性条件}{55}{7.39}}
