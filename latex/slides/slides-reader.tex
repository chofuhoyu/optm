% ============================================================
%  单元格尺寸：每页 2 张（上下各一）
%  可用区域约 116mm × 158mm
%  每格约 116mm × 76mm
% ============================================================
\newlength{\leafcellwidth}
\newlength{\leafcellheight}
\setlength{\leafcellwidth}{0.92\textwidth}
\setlength{\leafcellheight}{0.46\textheight}

% ============================================================
%  \refslide{文件名}{页码}{标签} — 老师幻灯片（单线框 + 右上角页码标签）
%  高度 = (textheight - fbox边框4×fboxrule - 4pt间距 - headroom) / 2
% ============================================================
\newlength{\refslideheadroom}
\setlength{\refslideheadroom}{16pt}
\newcommand{\refslide}[3]{%
  \centerline{%
    \fbox{%
      \begin{tikzpicture}[baseline=(img.south)]
        \node[inner sep=0] (img) {%
          \includegraphics[page=#2, width=\dimexpr\textwidth-2\fboxsep-2\fboxrule-2mm\relax, height=\dimexpr(\textheight-4\fboxrule-4pt-\refslideheadroom)/2\relax, keepaspectratio]%
            {../material/#1.pdf}%
        };
        \node[fill=black, text=white, font=\fontsize{12pt}{14pt}\selectfont\bfseries,
              inner sep=1.5pt, outer sep=0pt,
              anchor=north east] at (img.north east) {#3};
      \end{tikzpicture}%
    }%
  }%
}

% ============================================================
%  \emptycell — 空白单元格
% ============================================================
\newcommand{\emptycell}{%
  \begin{minipage}[t][\leafcellheight][c]{\leafcellwidth}
    \centering
  \end{minipage}%
}

% ============================================================
%  \ncell{内容} — 自己的笔记（双线框）
% ============================================================
\newcommand{\ncell}[1]{%
  \begin{minipage}[t][\leafcellheight][c]{\leafcellwidth}
    \centering
    \doublebox{%
      \begin{minipage}[t][\dimexpr\leafcellheight-4\fboxsep-4\fboxrule-2\doublerulesep\relax][c]{%
        \dimexpr\leafcellwidth-4\fboxsep-4\fboxrule-2\doublerulesep-4mm\relax%
      }
        \centering
        \vspace{0pt}%
        #1%
      \end{minipage}%
    }%
  \end{minipage}%
}

% ============================================================
%  \leafgridcell — 内部用：#1=文件名, #2=页码, #3=章节号
% ============================================================
\newcommand{\leafgridcell}[3]{%
  \ifnum#2=0
    \emptycell
  \else
    \refslide{#1}{#2}{#3.#2}%
  \fi
}

% ============================================================
%  \leafslidepair{文件}{p1}{p2}{章节号} — 一页两张幻灯片
%  页码填 0 表示空白
% ============================================================
\newcommand{\leafslidepair}[4]{%
  \vspace{1mm}%
  \noindent\leafgridcell{#1}{#2}{#4}\par
  \vspace{2mm}\noindent\leafgridcell{#1}{#3}{#4}\par
  \clearpage
}

% ============================================================
%  \leafnoteslide{内容} — 整页笔记（双线框）
% ============================================================
\newcommand{\leafnotepage}[1]{%
  \vspace{2mm}%
  \begin{center}
  \doublebox{%
    \begin{minipage}[t][0.92\textheight][c]{0.92\textwidth}
      \centering
      #1%
    \end{minipage}%
  }%
  \end{center}%
  \clearpage
}

% ============================================================
%  LaTeX3: 将 \slidesdata{ch}{list} 分组为每2个一页
%  \notedata{ch}{content} 包装为整页笔记
% ============================================================
\usepackage{expl3}
\ExplSyntaxOn
\int_new:N \l_slides_count_int
\tl_new:N \l_slides_prev_tl
\tl_new:N \l_chapter_num_tl

\cs_new:Npn \slidesdata #1#2 {
  % 从章节名提取章节号（如 "2数学基础" → "2"）
  \regex_extract_once:nnN {^(\d+)} {#1} \l_tmpa_seq
  \seq_pop:NN \l_tmpa_seq \l_tmpa_tl           % 丢弃完整匹配
  \seq_pop:NN \l_tmpa_seq \l_chapter_num_tl    % 捕获组1 = 章节号
  \clist_set:Nn \l_tmpa_clist {#2}
  \int_zero:N \l_slides_count_int
  \clist_map_inline:Nn \l_tmpa_clist {
    \int_incr:N \l_slides_count_int
    \int_compare:nNnTF { \int_mod:nn { \l_slides_count_int } { 2 } } = { 1 } {
      \tl_set:Nn \l_slides_prev_tl {##1}
    } {
      \leafslidepair{#1}{\l_slides_prev_tl}{##1}{\l_chapter_num_tl}
    }
  }
  \int_compare:nNnT { \int_mod:nn { \l_slides_count_int } { 2 } } = { 1 } {
    \leafslidepair{#1}{\l_slides_prev_tl}{0}{\l_chapter_num_tl}
  }
}

\cs_new:Npn \notedata #1#2 {
  \leafnotepage{#2}
}
\ExplSyntaxOff
