% !TeX program = xelatex
% ============================================================
%  Leaf5 专属排版 — 1680×1264 屏幕
%  每页 2 张幻灯片（上下排列），适合 e-ink 阅读
%  老师幻灯片 → 单线框 (\scell)
%  自己笔记   → 双线框 (\ncell)
% ============================================================
\documentclass[12pt]{ctexart}

% === 页面设置：适配 Leaf5 屏幕比例 ===
\usepackage[
  paperwidth=126.4mm,
  paperheight=168.0mm,
  margin=5mm
]{geometry}
\setlength{\parindent}{0pt}
\pagestyle{empty}

% === 宏包 ===
\usepackage{graphicx}
\usepackage{fancybox}
\usepackage{pgffor}
\usepackage{xcolor}
\usepackage{amsmath, amssymb}

% ============================================================
%  单元格尺寸：每页 2 张（上下各一）
%  可用区域约 116mm × 158mm
%  每格约 116mm × 76mm
% ============================================================
\newlength{\leafcellwidth}
\newlength{\leafcellheight}
\setlength{\leafcellwidth}{0.92\textwidth}
\setlength{\leafcellheight}{0.46\textheight}

% ============================================================
%  \scell{文件名}{页码} — 老师幻灯片（单线框）
% ============================================================
\newcommand{\scell}[2]{%
  \fbox{%
    \begin{minipage}[t][\leafcellheight][c]{\dimexpr\leafcellwidth-2\fboxsep-2\fboxrule\relax}
      \centering
      \vspace{0pt}%
      \includegraphics[%
        page=#2,%
        width=\dimexpr\leafcellwidth-2\fboxsep-2\fboxrule-6mm\relax,%
        height=\dimexpr\leafcellheight-2\fboxsep-2\fboxrule-3mm\relax,%
        keepaspectratio%
      ]{../material/#1.pdf}%
    \end{minipage}%
  }%
}

% ============================================================
%  \emptycell — 空白单元格
% ============================================================
\newcommand{\emptycell}{%
  \begin{minipage}[t][\leafcellheight][c]{\leafcellwidth}
    \centering
  \end{minipage}%
}

% ============================================================
%  \ncell{内容} — 自己的笔记（双线框）
% ============================================================
\newcommand{\ncell}[1]{%
  \begin{minipage}[t][\leafcellheight][c]{\leafcellwidth}
    \centering
    \doublebox{%
      \begin{minipage}[t][\dimexpr\leafcellheight-4\fboxsep-4\fboxrule-2\doublerulesep\relax][c]{%
        \dimexpr\leafcellwidth-4\fboxsep-4\fboxrule-2\doublerulesep-4mm\relax%
      }
        \centering
        \vspace{0pt}%
        #1%
      \end{minipage}%
    }%
  \end{minipage}%
}

% ============================================================
%  \leafgridcell — 内部用
% ============================================================
\newcommand{\leafgridcell}[2]{%
  \ifnum#2=0
    \emptycell
  \else
    \scell{#1}{#2}%
  \fi
}

% ============================================================
%  \leafslidepair{文件}{p1}{p2} — 一页两张幻灯片
%  页码填 0 表示空白
% ============================================================
\newcommand{\leafslidepair}[3]{%
  \vspace{1mm}%
  \noindent\leafgridcell{#1}{#2}\par
  \vspace{2mm}\noindent\leafgridcell{#1}{#3}\par
  \clearpage
}

% ============================================================
%  \leafnoteslide{内容} — 整页笔记（双线框）
% ============================================================
\newcommand{\leafnotepage}[1]{%
  \vspace{2mm}%
  \begin{center}
  \doublebox{%
    \begin{minipage}[t][0.92\textheight][c]{0.92\textwidth}
      \centering
      #1%
    \end{minipage}%
  }%
  \end{center}%
  \clearpage
}

% ============================================================
%  LaTeX3: 将 \slidesdata{ch}{list} 分组为每2个一页
%  \notedata{ch}{content} 包装为整页笔记
% ============================================================
\usepackage{expl3}
\ExplSyntaxOn
\int_new:N \l_slides_count_int
\tl_new:N \l_slides_prev_tl

\cs_new:Npn \slidesdata #1#2 {
  \clist_set:Nn \l_tmpa_clist {#2}
  \int_zero:N \l_slides_count_int
  \clist_map_inline:Nn \l_tmpa_clist {
    \int_incr:N \l_slides_count_int
    \int_compare:nNnTF { \int_mod:nn { \l_slides_count_int } { 2 } } = { 1 } {
      \tl_set:Nn \l_slides_prev_tl {##1}
    } {
      \leafslidepair{#1}{\l_slides_prev_tl}{##1}
    }
  }
  \int_compare:nNnT { \int_mod:nn { \l_slides_count_int } { 2 } } = { 1 } {
    \leafslidepair{#1}{\l_slides_prev_tl}{0}
  }
}

\cs_new:Npn \notedata #1#2 {
  \leafnotepage{#2}
}
\ExplSyntaxOff
