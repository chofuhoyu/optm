\notefullpage{
  \section*{第2章 数学基础 — 必背公式}

  \subsection*{梯度与 Hessian 矩阵}
  \begin{align*}
    \nabla f(x) &= \left[\frac{\partial f}{\partial x_1}, \dots, \frac{\partial f}{\partial x_n}\right]^T \\[4pt]
    \nabla^2 f(x) &= \begin{bmatrix}
      \frac{\partial^2 f}{\partial x_1^2} & \cdots & \frac{\partial^2 f}{\partial x_1 \partial x_n} \\
      \vdots & \ddots & \vdots \\
      \frac{\partial^2 f}{\partial x_n \partial x_1} & \cdots & \frac{\partial^2 f}{\partial x_n^2}
    \end{bmatrix}
  \end{align*}

  \subsection*{Taylor 展开}
  \begin{align*}
    f(x+d) = f(x) + \nabla f(x)^T d + \frac{1}{2} d^T \nabla^2 f(x) d + o(\|d\|^2)
  \end{align*}

  \subsection*{方向导数}
  \begin{align*}
    D_d f(x) = \lim_{t \to 0^+} \frac{f(x+td) - f(x)}{t} = \nabla f(x)^T d
  \end{align*}
}
